        EmpManager.go
        package main
        import(
          "fmt"
        )
        type Emp struct {
          Empno  int 
          Ename  string 
          Salary int 
        }
        func (emp Emp) print(){
          fmt.Println("Print EmpNo = " , emp.Empno , " , Ename =" , emp.Ename + ", Salary = " , emp.Salary)
        }
        type EmpManager struct {
          emparr []Emp 
        }
        func (empmgr *EmpManager) add(e1 Emp){
          empmgr.emparr = append(empmgr.emparr, e1)
        }
